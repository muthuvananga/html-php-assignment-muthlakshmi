<?php
function isLeapYear($year) {
    // Leap year if divisible by 400 or divisible by 4 but not by 100
    if (($year % 400 == 0) || (($year % 4 == 0) && ($year % 100 != 0))) {
        return true;
    } else {
        return false;
    }
}

// Example usage:
$year = 2024;
if (isLeapYear($year)) {
    echo "$year is a leap year.";
} else {
    echo "$year is not a leap year.";
}
?>


OUTPUT:
2024 is a leap year.
