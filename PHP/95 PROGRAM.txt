    $price = rand(10, 100); // Simulate price

        $_SESSION['cart'] = [];
    }

    $_SESSION['cart'][] = ['item' => $item, 'price' => $price];
    $message = "Item <strong>$item</strong> added to cart.";
}

// Handle clearing the cart
if (isset($_GET['clear'])) {
    unset($_SESSION['cart']);
    $message = "Cart has been cleared.";
