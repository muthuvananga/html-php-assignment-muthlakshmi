<?php
$email = '';
$validationResult = '';
$domain = '';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $email = trim($_POST['email'] ?? '');

    if (filter_var($email, FILTER_VALIDATE_EMAIL)) {
        $validationResult = "✅ Valid email address.";
        // Extract domain part
        $domain = substr(strrchr($email, "@"), 1);
    } else {
        $validationResult = "❌ Invalid email address.";
    }
}
?>

<!DOCTYPE html>
<html>
<head>
    <title>Email Validation</title>
</head>
<body>
    <h2>Email Validation and Domain Extraction</h2>

    <form method="POST" action="">
        <label for="email">Enter Email Address:</label><br>
        <input type="text" id="email" name="email" value="<?php echo htmlspecialchars($email); ?>" required>
        <br><br>
        <input type="submit" value="Validate">
    </form>

    <?php if ($_SERVER['REQUEST_METHOD'] === 'POST'): ?>
        <h3>Validation Result:</h3>
        <p><?php echo $validationResult; ?></p>

        <?php if ($domain): ?>
            <p>Domain Name: <strong><?php echo htmlspecialchars($domain); ?></strong></p>
        <?php endif; ?>
    <?php endif; ?>
</body>
</html>

Email Validation and Domain Extraction
Enter Email Address:
<?php echo htmlspecialchars($email); ?>
